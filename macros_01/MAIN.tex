\newcommand{\mymacro}[2]{
This is mymacro with arg1 = \texttt{#1} and arg2 = \texttt{#2}.
}

\DeclareRobustCommand{\mymacroTwo}[2]{
This is \texttt{\char`\\mymacroTwo} with arg1 = \texttt{#1} and arg2 = \texttt{#2}.
}
