\documentclass{article}

\usepackage{fancyvrb}
\usepackage{minted}
\usepackage{framed}

\newenvironment{markdown}%
    {\VerbatimEnvironment\begin{VerbatimOut}{tmp.markdown}}%
    {\end{VerbatimOut}%
        \immediate\write18{pandoc tmp.markdown -t latex -o tmp.tex}%
        \input{tmp}}

  \setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}
